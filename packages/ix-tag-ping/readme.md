# @onflow/ix-tag-ping

Exports a tag constant

# Usage

```javascript
// CommonJS
// npm install --save @onflow/ix-tag-ping
const {TAG: PING} = require("@onflow/ix-tag-ping")

// ESM
// npm install --save @onflow/ix-tag-ping
import {TAG as PING} from "@onflow/ix-tag-ping"

// Typescript
// npm install --save @onflow/ix-tag-ping
import {TAG as PING} from "@onflow/ix-tag-ping"

// Browser
<script type="module">
  import {TAG as PING} from "https://unpkg.com/@onflow/ix-tag-ping/tag.web.js"
</script>

// Deno
import {TAG as PING} from "https://unpkg.com/@onflow/ix-tag-ping/tag.deno.ts"
```
